Local variables

void MyFunction(){
	int a, b, c;

	a = 10;
	b = 5;
	c = 2;
}

Allocated memory for 3 integers (each 4 Byte): 3 * 4 = 12 Byte

_MyFunction:
	//allocate memory
	push ebp      
	mov ebp, esp  
	sub esp, 12   ; free space in stack for local variables (12 Bytes)
	
	//assign value
	mov [ebp - 4], 10   ; location of variable a
	mov [ebp - 8], 5    ; location of b
	mov [ebp - 12], 2   ; location of c

Standard Exit Sequence

Remove space for local variables, by reverting ESP to its old value (esp)

Return to the calling function with a ret command.

void myFunction3(int x, int y, int z){
	int a, int b, int c;

	return;
}
_MyFunction3:
	push ebp
	mov ebp, esp
	sub esp, 12 ; sizeof(a) + sizeof(b) + sizeof(c)
							; x = [ebp + 8], y = [ebp + 12], z = [ebp + 16]
							; a = [ebp - 4] = [esp + 8], b = [ebp - 8] = [esp + 4], c = [ebp - 12] = [esp]
	mov esp, ebp
	pop ebp
	ret 12       ; sizeof(x) + sizeof(y) + sizeof(z)
